CREATE TABLE lottery_results_NS (
    id INT AUTO_INCREMENT PRIMARY KEY,
    lottery_name VARCHAR(100) NOT NULL,
    draw_no VARCHAR(20),
    draw_date VARCHAR(50),
    results VARCHAR(200),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE lottery_results_NS_test (
    id INT AUTO_INCREMENT PRIMARY KEY,
    lottery_name VARCHAR(100) NOT NULL,
    draw_no VARCHAR(20),
    draw_date VARCHAR(50),
    results VARCHAR(200),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
